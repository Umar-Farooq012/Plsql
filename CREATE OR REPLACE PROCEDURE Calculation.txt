CREATE OR REPLACE PROCEDURE CalculateSum ( p_number1 IN NUMBER,  p_number2 IN NUMBER ) AS
v_sum NUMBER;
BEGIN
    -- Calculate the sum
    v_sum := p_number1 + p_number2;

    -- Display the result
    DBMS_OUTPUT.PUT_LINE('The sum of ' || p_number1 || ' and ' || p_number2 || ' is: ' || v_sum);
END;

DECLARE
    num1 NUMBER := 10;
    num2 NUMBER := 20;
BEGIN
    CalculateSum(num1, num2);
END;
